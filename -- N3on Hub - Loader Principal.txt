-- Detecta o jogo
local function detectGame()
	for _, id in ipairs(GAMES.BUILD_A_BOAT) do
		if gameId == id then
			return "Build a boat for treasure", "buildaboat"
		end
	end
	
	for _, id in ipairs(GAMES.BLOX_FRUITS) do
		if gameId == id then
			return "Blox Fruits", "bloxfruits"
		end
	end
	
	return "Universal", nil
end
